CREATE TABLE Region (
    ID_REGION INT PRIMARY KEY,
    NOM_REGION VARCHAR(30),
    IMG_REGION_URL VARCHAR(50),
);

CREATE TABLE Departement (
    ID_DEPARTEMENT INT PRIMARY KEY,
    NOM_DEPARTEMENT VARCHAR(30),
    ID_REGION INT,
    FOREIGN KEY (ID_REGION) REFERENCES Region(ID_REGION)
);

CREATE TABLE Utilisateur (
    ID_UTILISATEUR INT PRIMARY KEY,
    PSEUDO VARCHAR(30) UNIQUE,
    MOT_DE_PASSE_CHIFFRE VARCHAR(255) UNIQUE,
    MAIL VARCHAR(100) UNIQUE,
    ID_PHOTO INT,
    FOREIGN KEY (ID_PHOTO) REFERENCES Photo_Profile(ID_PHOTO)
);

CREATE TABLE Photo_Profile (
    ID_PHOTO INT PRIMARY KEY,
    URL_PHOTO VARCHAR(50)
);

CREATE TABLE Chat (
    ID_CHAT INT PRIMARY KEY,
    ID_DEPARTEMENT INT,
    PRIVE BOOLEAN,
    NOM_CHAT VARCHAR(50),
    FOREIGN KEY (ID_DEPARTEMENT) REFERENCES Departement(ID_DEPARTEMENT)
);

CREATE TABLE Message (
    ID_MESSAGE INT PRIMARY KEY,
    ID_CHAT INT,
    ID_UTILISATEUR INT,
    HEURE_ENVOIE TIME,
    TEXTE_ENVOIE TEXT,
    FOREIGN KEY (ID_CHAT) REFERENCES Chat(ID_CHAT),
    FOREIGN KEY (ID_UTILISATEUR) REFERENCES Utilisateur(ID_UTILISATEUR)
);

CREATE TABLE Chat_liked (
    ID_CHAT INT,
    ID_UTILISATEUR INT,
    PRIMARY KEY (ID_CHAT, ID_UTILISATEUR),
    FOREIGN KEY (ID_CHAT) REFERENCES Chat(ID_CHAT),
    FOREIGN KEY (ID_UTILISATEUR) REFERENCES Utilisateur(ID_UTILISATEUR)
);

CREATE TABLE Message_liked (
    ID_MESSAGE INT,
    ID_UTILISATEUR INT,
    PRIMARY KEY (ID_MESSAGE, ID_UTILISATEUR),
    FOREIGN KEY (ID_MESSAGE) REFERENCES Message(ID_MESSAGE),
    FOREIGN KEY (ID_UTILISATEUR) REFERENCES Utilisateur(ID_UTILISATEUR)
);
